Sample JSON Configuration

Here's an example of the JSON structure your endpoints file should follow: 

{
  "base_url": "https://api.example.com",
  "report_title": "API Load Test Report",
  "report_subtitle": "Performance testing for API endpoints",
  "tokens": [
    {
      "token": "your-auth-token-here",
      "name": "AUTH_TOKEN"
    }
  ],
  "endpoints": [
    {
      "name": "USER_LIST",
      "description": "Get list of users",
      "method": "GET",
      "url": "/api/v1/users",
      "headers": {
        "Authorization": "<AUTH_TOKEN>",
        "Content-Type": "application/json"
      },
      "weight": 30,
      "thinkTime": {
        "min": 1,
        "max": 3
      },
      "threshold_ms": 1000
    }
  ]
}
